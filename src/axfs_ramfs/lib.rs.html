1//! RAM filesystem used by [ArceOS](https://github.com/arceos-org/arceos).
2//!
3//! This crate provides a simple in-memory filesystem implementation based on
4//! [`axfs_vfs`]. It is fully stored in RAM and does not persist to disk,
5//! making it ideal for temporary files, testing, and scenarios where
6//! fast access is needed.
7//!
8//! # Components
9//!
10//! - [`RamFileSystem`] - The main filesystem structure implementing filesystem operations
11//! - [`DirNode`] - Directory node implementing directory operations
12//! - [`FileNode`] - File node implementing file operations
13//!
14//! # Features
15//!
16//! - Full support for file and directory operations
17//! - Hierarchical directory structure
18//! - In-memory storage with fast access
19//! - Compatible with the axfs_vfs interface
20//!
21//! [`axfs_vfs`]: axfs_vfs
22

40/// A RAM filesystem that implements VFS operations.
41///
42/// This is an in-memory filesystem that stores all data in RAM.
43/// It provides fast access but does not persist data across reboots.
44///
45/// # Fields
46///
47/// - `parent` - The parent filesystem mount point
48/// - `root` - The root directory of the RAM filesystem
49pub struct RamFileSystem {
50    parent: Once<VfsNodeRef>,
51    root: Arc<DirNode>,
52}
53
54impl RamFileSystem {
55    /// Create a new RAM filesystem instance.
56    ///
57    /// # Returns
58    ///
59    /// A new `RamFileSystem` with an empty root directory.
60    pub fn new() -> Self {
61        Self {
62            parent: Once::new(),
63            root: DirNode::new(None),
64        }
65    }
66
67    /// Returns the root directory node.
68    ///
69    /// # Returns
70    ///
71    /// A reference to the root directory of the filesystem.
72    pub fn root_dir_node(&self) -> Arc<DirNode> {
73        self.root.clone()
74    }
75}
76
77impl VfsOps for RamFileSystem {
78    /// Mount the RAM filesystem at the specified path.
79    ///
80    /// This method sets up the parent reference for the root directory.
81    ///
82    /// # Arguments
83    ///
84    /// * `_path` - The mount path (not used in RAM filesystem)
85    /// * `mount_point` - The mount point directory node
86    ///
87    /// # Returns
88    ///
89    /// Returns `Ok(())` on success.
90    fn mount(&self, _path: &str, mount_point: VfsNodeRef) -> VfsResult {
91        if let Some(parent) = mount_point.parent() {
92            self.root.set_parent(Some(self.parent.call_once(|| parent)));
93        } else {
94            self.root.set_parent(None);
95        }
96        Ok(())
97    }
98
99    /// Returns the root directory of the RAM filesystem.
100    ///
101    /// # Returns
102    ///
103    /// A reference to the root directory node.
104    fn root_dir(&self) -> VfsNodeRef {
105        self.root.clone()
106    }
107}
108
109impl Default for RamFileSystem {
110    /// Creates a default RAM filesystem instance.
111    ///
112    /// This is equivalent to calling [`RamFileSystem::new()`].
113    fn default() -> Self {
114        Self::new()
115    }
