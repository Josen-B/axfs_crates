11/// The directory node in RAM filesystem.
12///
13/// This represents a directory that can contain files and subdirectories.
14/// It implements the VFS node operations trait to provide directory operations.
15///
16/// # Fields
17///
18/// - `this` - Weak reference to self for creating child directories
19/// - `parent` - Weak reference to parent directory
20/// - `children` - Map of child node names to their references
21pub struct DirNode {
22    this: Weak<DirNode>,
23    parent: RwLock<Weak<dyn VfsNodeOps>>,
24    children: RwLock<BTreeMap<String, VfsNodeRef>>,
25}
26
27impl DirNode {
28    /// Creates a new directory node.
29    ///
30    /// # Arguments
31    ///
32    /// * `parent` - Optional weak reference to parent directory
33    ///
34    /// # Returns
35    ///
36    /// A new directory node wrapped in an Arc.
37    pub(super) fn new(parent: Option<Weak<dyn VfsNodeOps>>) -> Arc<Self> {
38        Arc::new_cyclic(|this| Self {
39            this: this.clone(),
40            parent: RwLock::new(parent.unwrap_or_else(|| Weak::<Self>::new())),
41            children: RwLock::new(BTreeMap::new()),
42        })
43    }
44
45    /// Sets the parent directory for this directory.
46    ///
47    /// # Arguments
48    ///
49    /// * `parent` - Optional reference to parent directory node
50    pub(super) fn set_parent(&self, parent: Option<&VfsNodeRef>) {
51        *self.parent.write() = parent.map_or(Weak::<Self>::new() as _, Arc::downgrade);
52    }
53
54    /// Returns a list of all entry names in this directory.
55    ///
56    /// # Returns
57    ///
58    /// A vector containing the names of all child nodes.
59    pub fn get_entries(&self) -> Vec<String> {
60        self.children.read().keys().cloned().collect()
61    }
62
63    /// Checks whether a node with the given name exists in this directory.
64    ///
65    /// # Arguments
66    ///
67    /// * `name` - The name to check for
68    ///
69    /// # Returns
70    ///
71    /// `true` if a node with the given name exists, `false` otherwise.
72    pub fn exist(&self, name: &str) -> bool {
73        self.children.read().contains_key(name)
74    }
75
76    /// Creates a new node with the given name and type in this directory.
77    ///
78    /// # Arguments
79    ///
80    /// * `name` - The name for the new node
81    /// * `ty` - The type of node to create (file or directory)
82    ///
83    /// # Returns
84    ///
85    /// Returns `Ok(())` if the node was created, or an error if creation failed.
86    ///
87    /// # Errors
88    ///
89    /// Returns [`VfsError::AlreadyExists`] if a node with the same name exists.
90    /// Returns [`VfsError::Unsupported`] if the node type is not supported.
91    pub fn create_node(&self, name: &str, ty: VfsNodeType) -> VfsResult {
92        if self.exist(name) {
93            log::error!("AlreadyExists {name}");
94            return Err(VfsError::AlreadyExists);
95        }
96        let node: VfsNodeRef = match ty {
97            VfsNodeType::File => Arc::new(FileNode::new()),
98            VfsNodeType::Dir => Self::new(Some(self.this.clone())),
99            _ => return Err(VfsError::Unsupported),
100        };
101        self.children.write().insert(name.into(), node);
102        Ok(())
103    }
104
105    /// Removes a node by the given name in this directory.
106    ///
107    /// Directories can only be removed if they are empty.
108    ///
109    /// # Arguments
110    ///
111    /// * `name` - The name of the node to remove
112    ///
113    /// # Returns
114    ///
115    /// Returns `Ok(())` if the node was removed, or an error otherwise.
116    ///
117    /// # Errors
118    ///
119    /// Returns [`VfsError::NotFound`] if the node does not exist.
120    /// Returns [`VfsError::DirectoryNotEmpty`] if attempting to remove a non-empty directory.
121    pub fn remove_node(&self, name: &str) -> VfsResult {
122        let mut children = self.children.write();
123        let node = children.get(name).ok_or(VfsError::NotFound)?;
124        if let Some(dir) = node.as_any().downcast_ref::<DirNode>() {
125            if !dir.children.read().is_empty() {
126                return Err(VfsError::DirectoryNotEmpty);
127            }
128        }
129        children.remove(name);
130        Ok(())
131    }
132}

134impl VfsNodeOps for DirNode {
135    /// Returns the attributes of this directory.
136    ///
137    /// # Returns
138    ///
139    /// Returns directory attributes with a fixed size of 4096 bytes.
140    fn get_attr(&self) -> VfsResult<VfsNodeAttr> {
141        Ok(VfsNodeAttr::new_dir(4096, 0))
142    }
143
144    /// Returns the parent directory of this directory.
145    ///
146    /// # Returns
147    ///
148    /// Returns `Some(VfsNodeRef)` if a parent exists, `None` otherwise.
149    fn parent(&self) -> Option<VfsNodeRef> {
150        self.parent.read().upgrade()
151    }
152
153    /// Lookups a node with the given path relative to this directory.
154    ///
155    /// This method supports path components including `.` (current directory)
156    /// and `..` (parent directory).
157    ///
158    /// # Arguments
159    ///
160    /// * `path` - The relative path to lookup
161    ///
162    /// # Returns
163    ///
164    /// Returns a reference to the found node, or an error if not found.
165    fn lookup(self: Arc<Self>, path: &str) -> VfsResult<VfsNodeRef> {
166        let (name, rest) = split_path(path);
167        let node = match name {
168            "" | "." => Ok(self.clone() as VfsNodeRef),
169            ".." => self.parent().ok_or(VfsError::NotFound),
170            _ => self
171                .children
172                .read()
173                .get(name)
174                .cloned()
175                .ok_or(VfsError::NotFound),
176        }?;
177
178        if let Some(rest) = rest {
179            node.lookup(rest)
180        } else {
181            Ok(node)
182        }
183    }
184
185    /// Reads directory entries into the provided buffer.
186    ///
187    /// The first two entries are always `.` (current directory) and
188    /// `..` (parent directory), followed by the actual child nodes.
189    ///
190    /// # Arguments
191    ///
192    /// * `start_idx` - The starting index for reading entries (for pagination)
193    /// * `dirents` - A mutable slice to store the directory entries
194    ///
195    /// # Returns
196    ///
197    /// Returns the number of entries read on success.
198    fn read_dir(&self, start_idx: usize, dirents: &mut [VfsDirEntry]) -> VfsResult<usize> {
199        let children = self.children.read();
200        let mut children = children.iter().skip(start_idx.max(2) - 2);
201        for (i, ent) in dirents.iter_mut().enumerate() {
202            match i + start_idx {
203                0 => *ent = VfsDirEntry::new(".", VfsNodeType::Dir),
204                1 => *ent = VfsDirEntry::new("..", VfsNodeType::Dir),
205                _ => {
206                    if let Some((name, node)) = children.next() {
207                        *ent = VfsDirEntry::new(name, node.get_attr().unwrap().file_type());
208                    } else {
209                        return Ok(i);
210                    }
211                }
212            }
213        }
214        Ok(dirents.len())
215    }
216
217    /// Creates a new node with the given path and type.
218    ///
219    /// This method recursively creates directories if needed.
220    ///
221    /// # Arguments
222    ///
223    /// * `path` - The path where the node should be created
224    /// * `ty` - The type of node to create
225    ///
226    /// # Returns
227    ///
228    /// Returns `Ok(())` if creation succeeds, or an error otherwise.
229    fn create(&self, path: &str, ty: VfsNodeType) -> VfsResult {
230        log::debug!("create {ty:?} at ramfs: {path}");
231        let (name, rest) = split_path(path);
232        if let Some(rest) = rest {
233            match name {
234                "" | "." => self.create(rest, ty),
235                ".." => self.parent().ok_or(VfsError::NotFound)?.create(rest, ty),
236                _ => {
237                    let subdir = self
238                        .children
239                        .read()
240                        .get(name)
241                        .ok_or(VfsError::NotFound)?
242                        .clone();
243                    subdir.create(rest, ty)
244                }
245            }
246        } else if name.is_empty() || name == "." || name == ".." {
247            Ok(()) // already exists
248        } else {
249            self.create_node(name, ty)
250        }
251    }
252
253    /// Removes a node at the given path.
254    ///
255    /// This method recursively removes nodes along the path.
256    /// Directories must be empty to be removed.
257    ///
258    /// # Arguments
259    ///
260    /// * `path` - The path of the node to remove
261    ///
262    /// # Returns
263    ///
264    /// Returns `Ok(())` if removal succeeds, or an error otherwise.
265    fn remove(&self, path: &str) -> VfsResult {
266        log::debug!("remove at ramfs: {path}");
267        let (name, rest) = split_path(path);
268        if let Some(rest) = rest {
269            match name {
270                "" | "." => self.remove(rest),
271                ".." => self.parent().ok_or(VfsError::NotFound)?.remove(rest),
272                _ => {
273                    let subdir = self
274                        .children
275                        .read()
276                        .get(name)
277                        .ok_or(VfsError::NotFound)?
278                        .clone();
279                    subdir.remove(rest)
280                }
281            }
282        } else if name.is_empty() || name == "." || name == ".." {
283            Err(VfsError::InvalidInput) // remove '.' or '..
284        } else {
285            self.remove_node(name)
286        }
287    }

292/// Splits a path into the first component and the rest.
293///
294/// This helper function is used for path resolution.
295///
296/// # Arguments
297///
298/// * `path` - The path to split
299///
300/// # Returns
301///
302/// A tuple containing the first path component and an optional remainder.
303fn split_path(path: &str) -> (&str, Option<&str>) {
304    let trimmed_path = path.trim_start_matches('/');
305    trimmed_path.find('/').map_or((trimmed_path, None), |n| {
306        (&trimmed_path[..n], Some(&trimmed_path[n + 1..]))
307    })
308}
