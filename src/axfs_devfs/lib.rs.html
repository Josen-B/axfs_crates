1//! Device filesystem used by [ArceOS](https://github.com/arceos-org/arceos).
2//!
3//! This crate provides a filesystem for managing device nodes, similar to
4//! `/dev` in Unix-like systems. It includes special device files such as
5//! null, zero, and urandom devices.
6//!
7//! The implementation is based on [`axfs_vfs`].
8//!
9//! # Components
10//!
11//! - [`DeviceFileSystem`] - The main device filesystem structure
12//! - [`DirNode`] - Directory node for device organization
13//! - [`NullDev`] - Null device (like `/dev/null`)
14//! - [`UrandomDev`] - Random number generator device (like `/dev/urandom`)
15//! - [`ZeroDev`] - Zero device (like `/dev/zero`)
16//!
17//! # Features
18//!
19//! - Static device registration (devices must be added at creation time)
20//! - Read-only directory structure (cannot create or remove devices dynamically)
21//! - Special device behaviors for null, zero, and random data
22

41/// A device filesystem that manages device nodes.
42///
43/// This filesystem provides access to special device files similar to
44/// `/dev` directory in Unix-like systems. Devices must be
45/// registered at filesystem creation time.
46///
47/// # Fields
48///
49/// - `parent` - The parent filesystem mount point
50/// - `root` - The root directory containing device nodes
51pub struct DeviceFileSystem {
52    parent: Once<VfsNodeRef>,
53    root: Arc<DirNode>,
54}
55
56impl DeviceFileSystem {
57    /// Creates a new device filesystem instance.
58    ///
59    /// # Returns
60    ///
61    /// A new device filesystem with an empty root directory.
62    pub fn new() -> Self {
63        Self {
64            parent: Once::new(),
65            root: DirNode::new(None),
66        }
67    }
68
69    /// Creates a subdirectory at the root directory.
70    ///
71    /// This method creates a new directory node and adds it to the root.
72    ///
73    /// # Arguments
74    ///
75    /// * `name` - The name of the subdirectory to create
76    ///
77    /// # Returns
78    ///
79    /// A reference to the created directory node.
80    pub fn mkdir(&self, name: &'static str) -> Arc<DirNode> {
81        self.root.mkdir(name)
82    }
83
84    /// Adds a device node to the root directory.
85    ///
86    /// This method registers a device node in the filesystem.
87    /// The node must implement the VFS node operations trait.
88    ///
89    /// # Arguments
90    ///
91    /// * `name` - The name of the device node
92    /// * `node` - The device node reference to add
93    pub fn add(&self, name: &'static str, node: VfsNodeRef) {
94        self.root.add(name, node);
95    }
96}
97
98impl VfsOps for DeviceFileSystem {
99    /// Mounts the device filesystem at the specified path.
100    ///
101    /// This method sets up the parent reference for the root directory.
102    ///
103    /// # Arguments
104    ///
105    /// * `_path` - The mount path (not used in device filesystem)
106    /// * `mount_point` - The mount point directory node
107    ///
108    /// # Returns
109    ///
110    /// Returns `Ok(())` on success.
111    fn mount(&self, _path: &str, mount_point: VfsNodeRef) -> VfsResult {
112        if let Some(parent) = mount_point.parent() {
113            self.root.set_parent(Some(self.parent.call_once(|| parent)));
114        } else {
115            self.root.set_parent(None);
116        }
117        Ok(())
118    }
119
120    /// Returns the root directory of the device filesystem.
121    ///
122    /// # Returns
123    ///
124    /// A reference to the root directory node.
125    fn root_dir(&self) -> VfsNodeRef {
126        self.root.clone()
127    }
128}

130impl Default for DeviceFileSystem {
131    /// Creates a default device filesystem instance.
132    ///
133    /// This is equivalent to calling [`DeviceFileSystem::new()`].
134    fn default() -> Self {
135        Self::new()
136    }
