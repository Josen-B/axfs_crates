4/// A urandom device behaves like `/dev/urandom`.
5///
6/// This device generates pseudo-random bytes using a Linear Congruential
7/// Generator (LCG) when read. It is useful for generating random data
8/// for testing and non-cryptographic purposes.
9///
10/// # Behavior
11///
12/// - Read operations: Return pseudo-random bytes
13/// - Write operations: Accept all data but discard it
14/// - Truncate operations: Always succeed with no effect
15///
16/// # Unix Equivalent
17///
18/// This device behaves similarly to `/dev/urandom` in Unix-like systems,
19/// though it uses a simple LCG and is not cryptographically secure.
20///
21/// # Algorithm
22///
23/// Uses a 64-bit LCG with the formula: `seed = seed * m + c`
24/// where `m = 6364136223846793005` and `c = 1`.
25pub struct UrandomDev {
26    seed: AtomicU64,
27}
28
29impl UrandomDev {
30    /// Creates a new urandom device with specified seed.
31    ///
32    /// # Arguments
33    ///
34    /// * `seed` - The initial seed value for the random number generator
35    ///
36    /// # Returns
37    ///
38    /// A new urandom device instance.
39    pub const fn new(seed: u64) -> Self {
40        Self {
41            seed: AtomicU64::new(seed),
42        }
43    }
44
45    /// Creates a new instance with a default seed.
46    ///
47    /// The default seed value is `0xa2ce_a2ce`.
48    ///
49    /// # Returns
50    ///
51    /// A new urandom device instance with default seed.
52    fn new_with_default_seed() -> Self {
53        Self::new(0xa2ce_a2ce)
54    }
55
56    /// Generates the next 64-bit pseudo-random number.
57    ///
58    /// This method uses a Linear Congruential Generator (LCG) algorithm:
59    /// `seed = seed * 6364136223846793005 + 1`
60    ///
61    /// # Returns
62    ///
63    /// The next pseudo-random 64-bit value.
64    fn next_u64(&self) -> u64 {
65        let new_seed = self
66            .seed
67            .load(Ordering::SeqCst)
68            .wrapping_mul(6364136223846793005)
69            + 1;
70        self.seed.store(new_seed, Ordering::SeqCst);
71        new_seed
72    }
73}
74
75impl Default for UrandomDev {
76    /// Creates a default urandom device instance.
77    ///
78    /// This is equivalent to calling [`UrandomDev::new_with_default_seed()`].
79    fn default() -> Self {
80        Self::new_with_default_seed()
81    }
82}

84impl VfsNodeOps for UrandomDev {
85    /// Returns attributes of the urandom device.
86    ///
87    /// # Returns
88    ///
89    /// Returns character device attributes with zero size.
90    fn get_attr(&self) -> VfsResult<VfsNodeAttr> {
91        Ok(VfsNodeAttr::new(
92            VfsNodePerm::default_file(),
93            VfsNodeType::CharDevice,
94            0,
95            0,
96        ))
97    }
98
99    /// Reads pseudo-random bytes from the device.
100    ///
101    /// This method fills the buffer with pseudo-random data.
102    /// The offset parameter is ignored as this device generates
103    /// fresh random data on each read.
104    ///
105    /// # Arguments
106    ///
107    /// * `_offset` - The read offset (ignored)
108    /// * `buf` - The buffer to fill with random bytes
109    ///
110    /// # Returns
111    ///
112    /// Always returns the buffer length.
113    fn read_at(&self, _offset: u64, buf: &mut [u8]) -> VfsResult<usize> {
114        for chunk in buf.chunks_mut(8) {
115            let random_value = self.next_u64();
116            let bytes = random_value.to_ne_bytes();
117            for (i, byte) in chunk.iter_mut().enumerate() {
118                if i < bytes.len() {
119                    *byte = bytes[i];
120                }
121            }
122        }
123        Ok(buf.len())
124    }
125
126    /// Writes to the urandom device.
127    ///
128    /// This operation discards all data.
129    ///
130    /// # Arguments
131    ///
132    /// * `_offset` - The write offset (ignored)
133    /// * `buf` - The buffer containing data to write (discarded)
134    ///
135    /// # Returns
136    ///
137    /// Always returns the buffer length (but data is discarded).
138    fn write_at(&self, _offset: u64, buf: &[u8]) -> VfsResult<usize> {
139        Ok(buf.len())
140    }
141
142    /// Truncates the urandom device (no effect).
143    ///
144    /// This operation always succeeds.
145    ///
146    /// # Arguments
147    ///
148    /// * `_size` - The truncation size (ignored)
149    ///
150    /// # Returns
151    ///
152    /// Always returns `Ok(())`.
153    fn truncate(&self, _size: u64) -> VfsResult {
154        Ok(())
155    }
156
157    axfs_vfs::impl_vfs_non_dir_default! {}
158}
