3/// A zero device behaves like `/dev/zero`.
4///
5/// This device always returns null bytes (`\0`) on reads and discards
6/// all written data. It is commonly used for creating zero-filled
7/// files or buffers.
8///
9/// # Behavior
10///
11/// - Read operations: Always return buffer filled with null bytes
12/// - Write operations: Accept all data but discard it
13/// - Truncate operations: Always succeed with no effect
14///
15/// # Unix Equivalent
16///
17/// This device behaves similarly to `/dev/zero` in Unix-like systems.
18pub struct ZeroDev;
19
20impl VfsNodeOps for ZeroDev {
21    /// Returns attributes of the zero device.
22    ///
23    /// # Returns
24    ///
25    /// Returns character device attributes with zero size.
26    fn get_attr(&self) -> VfsResult<VfsNodeAttr> {
27        Ok(VfsNodeAttr::new(
28            VfsNodePerm::default_file(),
29            VfsNodeType::CharDevice,
30            0,
31            0,
32        ))
33    }
34
35    /// Reads from the zero device.
36    ///
37    /// This operation fills the entire buffer with null bytes.
38    ///
39    /// # Arguments
40    ///
41    /// * `_offset` - The read offset (ignored)
42    /// * `buf` - The buffer to fill with null bytes
43    ///
44    /// # Returns
45    ///
46    /// Always returns the buffer length.
47    fn read_at(&self, _offset: u64, buf: &mut [u8]) -> VfsResult<usize> {
48        buf.fill(0);
49        Ok(buf.len())
50    }
51
52    /// Writes to the zero device.
53    ///
54    /// This operation discards all data.
55    ///
56    /// # Arguments
57    ///
58    /// * `_offset` - The write offset (ignored)
59    /// * `buf` - The buffer containing data to write (discarded)
60    ///
61    /// # Returns
62    ///
63    /// Always returns the number of bytes written (but data is discarded).
64    fn write_at(&self, _offset: u64, buf: &[u8]) -> VfsResult<usize> {
65        Ok(buf.len())
66    }
67
68    /// Truncates the zero device (no effect).
69    ///
70    /// This operation always succeeds.
71    ///
72    /// # Arguments
73    ///
74    /// * `_size` - The truncation size (ignored)
75    ///
76    /// # Returns
77    ///
78    /// Always returns `Ok(())`.
79    fn truncate(&self, _size: u64) -> VfsResult {
80        Ok(())
81    }
82
83    axfs_vfs::impl_vfs_non_dir_default! {}
84}

90    #[test]
91    fn test_zero_dev_get_attr() {
92        let zero = ZeroDev;
93        let attr = zero.get_attr().unwrap();
94        assert_eq!(attr.file_type(), VfsNodeType::CharDevice);
95        assert_eq!(attr.size(), 0);
96        assert_eq!(attr.blocks(), 0);
97    }
98
99    #[test]
100    fn test_zero_dev_read() {
101        let zero = ZeroDev;
102        let mut buf = [1; 100];
103        let read = zero.read_at(0, &mut buf).unwrap();
104        assert_eq!(read, 100);
105        // Buffer should be filled with zeros
106        assert_eq!(buf, [0; 100]);
107    }
108
109    #[test]
110    fn test_zero_dev_read_offset() {
111        let zero = ZeroDev;
112        let mut buf = [1; 50];
113        let read = zero.read_at(100, &mut buf).unwrap();
114        assert_eq!(read, 50);
115        // Offset is ignored, buffer should still be filled with zeros
116        assert_eq!(buf, [0; 50]);
117    }
118
119    #[test]
120    fn test_zero_dev_read_empty() {
121        let zero = ZeroDev;
122        let mut buf = [];
123        let read = zero.read_at(0, &mut buf).unwrap();
124        assert_eq!(read, 0);
125    }

127    #[test]
128    fn test_zero_dev_write() {
129        let zero = ZeroDev;
130        let data = b"Hello, World!";
131        let written = zero.write_at(0, data).unwrap();
132        assert_eq!(written, data.len());
133        // Data is discarded
134    }

146        let zero = ZeroDev;
